\newglossaryentry{Issue}{
	name = Issue,
	description = {Arbeitspaket (\emph{Projektmanagement}),
		Problem (\emph{Umgangssprache}).
		Eine Problemstellung die zu lösen ist oder zur
		Diskussion freisteht
		(\emph{GitHub})}
}

\newglossaryentry{Milestone}{
	name = Milestone,
	description = {Meilenstein -- Ereignis besonderer Bedeutung
		(\emph{Projektmanagement}).
		Termin eines Projektes, zu welchem zugehörige
		Arbeitspakete abzuschliessen sind (\emph{GitHub})}
}

\newglossaryentry{Title}{
	name = Title,
	description = {Titel (\emph{englisch}).
		Eine kurzbeschreibung für ein Arbeitspaket
		oder Meilenstein (\emph{GitHub})}
}

\newglossaryentry{Comment}{
	name = Comment,
	description = {Kommentar (\emph{englisch}).
		Kurzbeschreibeung oder Foreneintrag (\emph{GitHub})}
}

\newglossaryentry{Event}{
	name = Event,
	description = {Ereigniss (\emph{englisch}).
		Änderung [eines Attributs] (\emph{GitHub, Issue})}
}

\newglossaryentry{Label}{
	name = Label,
	description = {Markierung oder Etikett (\emph{englisch}).
		Ein Attribut zur Klassifizierung (\emph{GitHub})}
}
